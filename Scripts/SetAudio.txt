set alreadyRunning to 0set selectedPane to missing valuetell application "System Events"	if (get name of every application process) contains "System Preferences" then		set alreadyRunning to 1	end ifend telltell application "System Preferences"	if alreadyRunning is 1 then		set selectedPane to current pane	end if	reveal anchor "[[inout]]" of pane id "com.apple.preference.sound"	tell application "System Events"		repeat until (exists window "Sound" of application process "System Preferences")			delay 0.025		end repeat		set windowLoaded to 0		repeat until windowLoaded is 1			set aTitle to get title of window 1 of application process "System Preferences"			if "..." is not in aTitle then				set windowLoaded to 1			end if			delay 0.025		end repeat		# Wait until we have the proper elements		repeat until (exists tab group 1 of window "Sound" of application process "System Preferences")			delay 0.025		end repeat		tell tab group 1 of window "Sound" of application process "System Preferences"			set selected of row [[row]] of table 1 of scroll area 1 to true		end tell		if alreadyRunning is 1 then			if selectedPane is not missing value then				tell application "System Preferences" to set current pane to selectedPane			else				tell application "System Preferences" to set show all to true			end if		end if	end tell	if alreadyRunning is 0 then		quit	end ifend tell
