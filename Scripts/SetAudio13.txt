set alreadyRunning to 0set selectedPane to missing valuetell application "System Events"	if (get name of every application process) contains "System Settings" then		set alreadyRunning to 1	end ifend telltell application "System Settings"	if alreadyRunning is 1 then		set selectedPane to current pane	end if	reveal anchor "[[inout]]" of pane id "com.apple.Sound-Settings.extension"	tell application "System Events"		repeat until (exists window "Sound" of application process "System Settings")			delay 0.025		end repeat		set windowLoaded to 0		repeat until windowLoaded is 1			set aTitle to get title of window 1 of application process "System Settings"			if "..." is not in aTitle then				set windowLoaded to 1			end if			delay 0.025		end repeat		# Wait until we have the proper elements		repeat until (exists outline 1 of scroll area 1 of group 2 of scroll area 1 of group 1 of group 2 of splitter group 1 of group 1 of window "Sound" of application process "System Settings")			delay 0.025		end repeat
		# Set the value		tell outline 1 of scroll area 1 of group 2 of scroll area 1 of group 1 of group 2 of splitter group 1 of group 1 of window "Sound" of application process "System Settings" to set selected of row [[row]] to true		if alreadyRunning is 1 then			if selectedPane is not missing value then				tell application "System Settings" to set current pane to selectedPane			else				tell application "System Settings" to set show all to true			end if		end if	end tell	if alreadyRunning is 0 then		quit	end ifend tell
